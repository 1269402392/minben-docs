<template>
    <div class="SchemaMetadataList">
        <rdapp-table :data="metadataList" :tableColumns="tableColumns" :showSelection="false" :pageOptions="null">
            <template #viewEtlDetail="{ row }">
                <el-button type="primary" @click="viewEtlDetail(row)">跳转详情</el-button>
            </template>
        </rdapp-table>
    </div>
</template>

<script setup>
import services from '@/services';

const { $message } = inject('globalProperties');

const props = defineProps({
    currentMetadata: { type: Object, required: true },
});

const tableColumns = [
    { label: '元数据名称', prop: 'name' },
    { label: '描述', prop: 'description' },
    { label: '操作', prop: 'viewEtlDetail', width: 220 },
];

const metadataList = ref([]);

// 获取当前表关联作业列表
const getMetadataList = async (page = 1, perPage = 10) => {
    const { bindSchema, bindTable } = props.currentMetadata ?? {};
    const params = {
        databaseName: bindSchema,
        tableName: bindTable,
        page,
        per_page: perPage,
    };
    const { records = [], hasError, status = -1 } = await services.dasBiz.jobMetaSearch(params);
    if (hasError) {
        $message.error(`${status}: 获取元数据列表失败`);
        return;
    }
    metadataList.value = records;
};
getMetadataList();

const viewEtlDetail = row => {
    const { protocol, hostname } = location;
    const url = `${protocol}//${hostname}:10009/metadataViewdData?isMqType=${row.dataType === 10}&id=${row.id}`;
    window.open(url);
};
</script>

js
import { reactive, toRefs, toValue } from 'vue';
import { ElMessage } from 'element-plus';
import services from '@/services';
import { ADD, EDIT, DELETE } from '@/utils/constant';

function getBizService(fetch) {
    if (typeof fetch === 'function') {
        return fetch;
    }
    const [module, func] = fetch.split('.');
    return services[module][func].bind(services[module]);
}

/**
 * @description 表格数据处理
 * @param fetch {Function | String} 请求方法(function | 'aggs.searchCameras')
 * @param type {String} 请求类型
 * @param params {Object} 请求参数(biz请求可传) ref
 * @param errMsg {String} 错误提示信息
 */
function useTable(fetch, { type = 'post', params = {}, errMsg } = {}) {
    const groupName = 'bizUseTable';
    const state = reactive({
        tableData: [], // 表格数据
        totalCount: 0, // 表格总数
        pageIndex: 1, // 当前页码
        pageSize: 25, // 每页显示条数
        tableLoading: false, // 加载状态
        tabletSelection: [], // 选中行
        isInitTable: false, // 是否初始化表格
    });

    /** 获取表格数据 */
    const getTableData = async () => {
        state.tableLoading = true;
        const sendParams = {
            [type === 'post' ? 'pageIndex' : 'page']: state.pageIndex,
            [type === 'post' ? 'pageSize' : 'per_page']: state.pageSize,
            ...toValue(params),
        };
        services.cancel(groupName); // 取消请求
        const {
            records = [],
            totalCount: total = 0,
            hasError = false,
            status,
        } = await getBizService(fetch)(sendParams, { groupName });
        hasError && ElMessage.error(status ? `${status}: ${errMsg || '获取数据失败'}` : errMsg || '获取数据失败');
        state.tableData = records;
        state.totalCount = total;
        state.tableLoading = false;
        state.isInitTable = true;
    };

    const pageChange = pageIndex => {
        state.pageIndex = pageIndex;
        getTableData();
    };

    /** 重新计算页码 */
    const reloadPageSize = (type, pageIndex) => {
        const map = {
            [ADD]: () => {
                state.totalCount += 1;
                state.pageIndex = pageIndex ? pageIndex : Math.ceil(map[type] / state.pageSize);
            },
            [DELETE]: () => {
                if (state.tableData.length !== 1) return;
                state.pageIndex = state.pageIndex > 1 ? state.pageIndex - 1 : 1;
            },
        };
        if (!type || type === EDIT || !map[type]) return;
        map[type]();
    };

    return {
        ...toRefs(state),
        getTableData,
        pageChange,
        reloadPageSize,
    };
}

export { useTable };

vue
<template>
    <el-table
        v-bind="$attrs"
        v-no-data="data?.length"
        :data="data"
        class="rdappTable"
        show-overflow-tooltip
        stripe
    >
        <el-table-column type="selection" v-if="tableColumns.length && showSelection" />
        <template v-for="item in tableColumns" :key="item.prop">
            <el-table-column v-bind="item" class-name="row-item">
                <template #header>
                    <slot :name="`${item.prop}-header`">{{ item.label }}</slot>
                </template>
                <template #default="scope">
                    <slot :name="item.prop" v-bind="scope">{{ scope.row[item.prop] || '-' }}</slot>
                </template>
            </el-table-column>
        </template>
        <el-table-column
            v-if="tableColumns.length && handlerOptions.show"
            class-name="handler"
            align="center"
            :show-overflow-tooltip="false"
            :width="handlerOptions.width || ''"
        >
            <template #header>
                <div class="handler">操作</div>
            </template>
            <template #default="scope">
                <slot name="handler">
                    <template v-for="(item, index) in handlerOptions.list" :key="index">
                        <el-button
                            v-if="['primary', 'warning'].includes(item.type)"
                            type="primary"
                            @click="item.handler(scope.row)"
                        >
                            {{ item.label }}
                        </el-button>
                        <rdapp-tooltip
                            v-else-if="item.type.startsWith('icon')"
                            :disabled="!item.tooltip"
                            disableAutoShow
                            :content="item.tooltip"
                        >
                            <em :class="['iconfont', item.type]" @click="item.handler(scope.row)"></em>
                        </rdapp-tooltip>
                        <el-popconfirm
                            v-else-if="item.type === 'alert'"
                            confirm-button-text="确定"
                            cancel-button-type="info"
                            cancel-button-text="取消"
                            title="确定删除吗 ?"
                            @confirm="item.handler(scope.row)"
                        >
                            <template #reference>
                                <div v-if="typeof item.inner === 'string'" v-html="item.inner"></div>
                                <component :is="item.inner" @click="item.handler(scope.row)" />
                            </template>
                        </el-popconfirm>
                    </template>
                </slot>
            </template>
        </el-table-column>
    </el-table>
    <el-pagination
        v-if="pageOptions"
        background
        layout="total, prev, pager, next"
        :total="pageOptions.totalCount"
        :page-size="pageOptions.pageSize"
        :current-page="pageOptions.pageIndex"
        :hide-on-single-page="$attrs['hide-on-single-page'] || false"
        @current-change="$emit('current-change', $event)"
    >
    </el-pagination>
</template>

<script setup>
defineOptions({
    inheritAttrs: false,
});

const props = defineProps({
    data: {
        type: Array,
        required: true,
    },
    tableColumns: {
        type: Array,
        required: true,
    },
    pageOptions: {
        type: Object,
        default: () => ({
            pageIndex: 1,
            pageSize: 10,
            totalCount: 110,
        }),
    },
    showSelection: {
        type: Boolean,
        default: true,
    },
    // 操作列配置
    handlerOptions: {
        type: Object,
        default: () => ({ show: false, list: [] }),
    },
});

defineEmits(['current-change']);
</script>

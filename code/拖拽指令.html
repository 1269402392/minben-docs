let zIndex = 1000; // 全局层级

// ios 的 safari 不支持该属性，兼容处理
const setCursor = (el, cursor) => {
    try {
        el.style.cursor = cursor;
    } catch (err) {
        console.warn('setCursor ~ cursor', '浏览器不支持 cursor 属性');
    }
};

// 解析matrix的正则
const matrix3dReg = /^matrix3d\((?:[-\d.]+,\s*){12}([-\d.]+),\s*([-\d.]+)(?:,\s*[-\d.]+){2}\)/
const matrixReg = /^matrix\((?:[-\d.]+,\s*){4}([-\d.]+),\s*([-\d.]+)\)$/


// 解析transform | transform3d的值
function getTransformValue(transform) {
    const transformArr = transform.match(matrix3dReg) ?? transform.match(matrixReg);
    return [Number(transformArr[1]), Number(transformArr[2])];
}

// 设置上下左右最大偏移量
function getBoundary(el, targetArea) {
    const { width, height } = el.getBoundingClientRect();
    const { offsetLeft, offsetTop } = el
    const { clientWidth, clientHeight } = targetArea ? document.querySelector(targetArea) : document.body;
    console.log(`🚀 ~ getBoundary ~ clientWidth-> `, clientWidth)
    return {
        maxLeft: 0 - offsetLeft,
        maxRight: clientWidth - width - offsetLeft,
        maxTop: 0 - offsetTop,
        maxBottom: clientHeight - height - offsetTop
    }
}

function init(el, value) {
    const { controlDom, unavailable, targetArea } = value ?? {}
    // 如果拖拽事件为不可用时不执行下面方法
    if (unavailable) {
        return;
    }
    //  如果有传入指定的元素控制，就用传入的元素，如果没有，就用第一个元素
    const dragDomObject = controlDom ? dragDom.querySelector(controlDom) : el.firstChild;
    // 设置鼠标手势
    setCursor(dragDomObject, 'grab');

    // 注册mousedown事件
    dragDomObject.onmousedown = mousedownEvent => {
        // 层级+1
        zIndex += 1;
        // 设置层级
        el.style.zIndex = zIndex;
        // 当按下时候，设置动画过度为none，既为及时渲染
        el.style.transition = 'none';
        // 设置鼠标手势为抓住
        setCursor(dragDomObject, 'grabbing');

        let translateX = 0
        let translateY = 0
        // 获取拖拽元素的transform偏移量
        const { transform } = getComputedStyle(el);
        if (transform !== 'none') {
            [translateX, translateY] = getTransformValue(transform);
        }

        const { maxLeft, maxBottom, maxRight, maxTop } = getBoundary(el, targetArea);
        console.log(`🚀 ~ maxLeft: ${maxLeft}, maxRight: ${maxRight}, maxTop: ${maxTop}, maxBottom: ${maxBottom}`);
        // 鼠标移动
        document.onmousemove = mousemoveEvent => {
            // 获取鼠标移动的距离
            let moveX = mousemoveEvent.clientX - mousedownEvent.clientX + translateX;
            let moveY = mousemoveEvent.clientY - mousedownEvent.clientY + translateY;

            // 边界检测
            moveX = Math.min(Math.max(maxLeft, moveX), maxRight);
            moveY = Math.min(Math.max(maxTop, moveY), maxBottom);

            // 设置元素的位置(移动的距离 + transform的偏移量, 使用z轴为了让浏览器开启GPU加速)
            el.style.transform = `translate3d(${moveX}px, ${moveY}px, 1px)`;
        };

        // 鼠标抬起事件
        document.onmouseup = () => {
            // 如果拖拽事件为不可用时不执行下面方法
            if (unavailable) return;
            // 鼠标抬起时，恢复鼠标手势为抓手
            setCursor(dragDomObject, 'grab');
            // 取消鼠标事件
            document.onmousemove = null;
            document.onmouseup = null;
            // 取消过度动画为none
            el.style.transition = '';
        };
    }
}

/**
 * @description 拖拽指令
 * @param {Boolean} value.unavailable 拖拽是否不可用
 * @param {String} value.controlDom 指定拖拽的元素
 * @param {String} targetArea 控制目标区域，不传为body区域内
 */
const drag = {
    mounted(el, { value = {} }) {
        zIndex += 1;
        init(el, value);
    },
    updated(el, { value }) {
        init(el, value, true);
    }
};

export default drag;

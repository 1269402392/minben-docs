tooltip溢出显示 ​

使用 ​

Vue
<div v-el-tooltip-auto-show="{ disable: false }">
    <el-tooltip v-bind="$attrs">
        <slot></slot>
    </el-tooltip>
</div>
javascript
/**
 * @description: el-tooltip溢出隐藏时才显示tooltip
 * @param {disable} 是否禁用, 如果禁用则跟默认效果一样
 */
const elTooltipAutoShow = {
    mounted(el, { value = {} }, vnode) {
        const { disable = false } = value;
        if (disable) return;
        const tooltipNode = vnode.children.find(c => c.component?.type.name === 'ElTooltip');
        if (tooltipNode) {
            const callback = throttle(() => {
                tooltipNode.component.props.disabled = true;
                const range = document.createRange();
                range.setStart(el, 0);
                range.setEnd(el, el.childNodes.length);
                const rangeWidth = Math.round(range.getBoundingClientRect().width);
                const padding =
                    (parseInt(getStyle(el, 'paddingLeft'), 10) || 0) +
                    (parseInt(getStyle(el, 'paddingRight'), 10) || 0);
                if (rangeWidth + padding > el.offsetWidth || el.scrollWidth > el.offsetWidth) {
                    tooltipNode.component.props.disabled = false;
                }
            }, 50);
            el.callback = callback;
            el.addEventListener('mouseenter', callback);
        }
    },
    unmounted(el) {
        if (!el.callback) return;
        el.removeEventListener('mouseenter', el.callback);
    },
};

function getStyle(element, styleName) {
    try {
        const computed = getComputedStyle(element, '');
        return element.style[styleName] || computed ? computed[styleName] : null;
    } catch (e) {
        return element.style[styleName];
    }
}
